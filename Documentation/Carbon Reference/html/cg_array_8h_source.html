651  // Name : _realloc ()
656  //-------------------------------------------------------------------------
657  void _realloc( size_type newSize, size_type newCapacity, size_type operateAt )
658  {
659  // How many entries are being inserted (0 if equal or shrinking)
660  difference_type insertCount = (newSize > mSize) ? newSize - mSize : 0;
661 
662  // How many entries are being removed (0 if equal or growing)
663  difference_type removeCount = (newSize < mSize) ? mSize - newSize : 0;
664 
665  // If size isn't changing, just assume we're operating at the end
666  // to simplify later sections of this method.
667  if ( newSize == mSize )
668  operateAt = mSize;
669 
670  // Clamp the operation point to the NEW size (in case of shrinking).
671  if ( operateAt > newSize )
672  operateAt = newSize;
673 
674  // If existing capacity differs from the prior capacity
675  // then we need to re-allocate.
676  if ( newCapacity != mCapacity )
677  {
678  // Allocate enough room in a new buffer for new capacity (not
679  // initialized at this stage). Take care to include the 'end'
680  // element that always exists at the end of the array.
681  char * newBuffer = new char[ (newCapacity + 1) * sizeof(value_type) ];
682 
683  // Any prior data to copy?
684  if ( mStart )
685  {
686  // We now need to duplicate the data from the old buffer. This
687  // is achieved as a two stage process for the data that falls
688  // on either side of the element specified by the 'operateAt'
689  // parameter. We use placement new to invoke the appropriate copy
690  // constructor in all cases since the new buffer's memory has
691  // yet to be initialized.
692  pointer dst = ((pointer)newBuffer) + operateAt;
693  pointer last = ((pointer)newBuffer);
694  pointer src = ((pointer)mStart) + operateAt;
695  while ( dst != last )
696  new (--dst) value_type( *--src );
697 
698  // Now do the same for the other side of the point of operation.
699  // We potentially need to move data after the insertion point 'up' by
700  // the required number of places indicated by the difference in size
701  // (unless we are shrinking, or inserting at the end of the array).
702  dst = ((pointer)newBuffer) + newSize;
703  last = ((pointer)newBuffer) + operateAt + insertCount;
704  src = ((pointer)mEnd);
705  while ( dst != last )
706  new (--dst) value_type( *--src );
707 
708  } // End if has old data
709 
710  // This is now our new buffer
711  _dealloc();
712  mStart = (storage_type)newBuffer;
713 
714  } // End if capacity altered
715  else
716  {
717  // Growing or shrinking?
718  if ( insertCount > 0 && operateAt < mSize )
719  {
720  // We potentially need to move data after the insertion point 'up' by
721  // the required number of places indicated by the difference in size.
722  // For all entries that have yet to be constructed, we need to use
723  // placement new and the copy constructor.
724  pointer dst = (((pointer)mStart)+newSize);
725  pointer last = (dst-insertCount);
726  pointer src = last;
727  while ( dst != last )
728  new ( --dst ) value_type( *--src );
729 
730  // We can just use the assignment operator for the rest.
731  last = (((pointer)mStart)+operateAt+insertCount);
732  while ( dst != last )
733  *--dst = *--src;
734 
735  // Finally; destruct the values that exist between the insert
736  // range that have since been moved by the above loop(s).
737  last = (((pointer)mStart)+operateAt);
738  while ( dst != last )
739  (--dst)->~value_type();
740 
741  } // End if inserting
742  else if ( removeCount > 0 )
743  {
744  // We are potentially shrinking. First ensure that the destructor
745  // is called for any elements in the buffer that are positioned
746  // within the region to be removed.
747  pointer dst = ((pointer)mStart)+operateAt+removeCount;
748  pointer last = (dst-removeCount);
749  while ( dst != last )
750  (--dst)->~value_type();
751 
752  // Shift values down as necessary using placement new
753  // to reconstruct the values, ensuring we call the
754  // destructor on each prior element as it moves.
755  dst = ((pointer)mStart)+operateAt;
756  pointer src = dst + removeCount;
757  last = ((pointer)mStart)+mSize;
758  while ( src != last )
759  {
760  new ( dst++ ) value_type( *src );
761  (src++)->~value_type();
762  }
763 
764  } // End if removing
765 
766  } // End if no capacity alteration
767 
768  // Resize complete.
769  mSize = newSize;
770  mCapacity = newCapacity;
771  mEnd = (storage_type)(((pointer)mStart)+mSize);
772  }
