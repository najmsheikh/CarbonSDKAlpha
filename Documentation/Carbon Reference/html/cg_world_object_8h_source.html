11 // Name : cgWorldObject.h //
12 // //
13 // Desc : Base class for all objects that make up a scene. Objects are //
14 // basically the foundation of all scene components such as meshes, //
15 // cameras, light sources, etc. Objects don't specifically maintain //
16 // scene information such as transformation matrices since they //
17 // technically belong to the world and can exist in multiple scenes. //
18 // Objects can however be referenced by object node types to provide //
19 // their final connection to the scene(s) as required (i.e. //
20 // cgObjectNode and its derived types to give them position, //
21 // orientation, etc.) //
22 // //
