271 class cgSceneUpdateFIFO
272 {
273 public:
274  //-------------------------------------------------------------------------
275  // Constructors & Destructors
276  //-------------------------------------------------------------------------
277  cgSceneUpdateFIFO( size_t size )
278  {
279  mCount = 0;
280  mCurrent = 0;
281  mBottom = 0;
282  mSize = size;
283  mBuffer = new cgObjectNode*[size];
284  }
285 
286  ~cgSceneUpdateFIFO( )
287  {
288  clear();
289  }
290 
291  //-------------------------------------------------------------------------
292  // Public Methods
293  //-------------------------------------------------------------------------
294  cgObjectNode ** push( cgObjectNode * node )
295  {
296  ++mCount;
297  cgObjectNode ** result = &mBuffer[mCurrent];
298  mBuffer[mCurrent] = node;
299  ++mCurrent;
300  if ( mCurrent == mSize )
301  mCurrent = 0;
302  return result;
303  };
304 
305  cgObjectNode * pop()
306  {
307  while ( mCurrent != mBottom )
308  {
309  --mCount;
310  cgObjectNode * result = mBuffer[mBottom];
311  ++mBottom;
312  if ( mBottom == mSize )
313  mBottom = 0;
314  if ( result )
315  return result;
316  }
317  return CG_NULL;
318  }
319 
320  bool flush( cgObjectNode * node )
321  {
322  if ( mCurrent == mBottom )
323  return false;
324 
325  for ( size_t i = mBottom; i != mCurrent; )
326  {
327  if ( mBuffer[i] == node )
328  {
329  mBuffer[i] = CG_NULL;
330  return true;
331  }
332  if ( ++i == mSize )
333  i = 0;
334 
335  } // Next
336  return false;
337  }
338 
339  //-------------------------------------------------------------------------
340  // Name : clear ()
344  //-------------------------------------------------------------------------
345  void clear( )
346  {
347  delete []mBuffer;
348  mCount = 0;
349  mCurrent = 0;
350  mBottom = 0;
351  mSize = 0;
352  mBuffer = CG_NULL;
353  }
354 
355  //-------------------------------------------------------------------------
356  // Name : getEntryCount ()
360  //-------------------------------------------------------------------------
361  inline size_t getEntryCount( ) const
362  {
363  return mCount;
364  }
365 
366 private:
367  //-------------------------------------------------------------------------
368  // Private Variables
369  //-------------------------------------------------------------------------
370  size_t mCount;
371  size_t mCurrent;
372  size_t mBottom;
373  size_t mSize;
374  cgObjectNode ** mBuffer;
375 };
