270  //-------------------------------------------------------------------------
271  inline _Vt& operator=( const _Vt& value )
272  {
273  if ( _data->refCount == 1 )
274  {
275  _data->value = value;
276 
277  } // End if not shared
278  else
279  {
280  Data * newData = new Data(value);
281  if ( --_data->refCount == 0 )

285  } // End if shared

287  }

299  //-------------------------------------------------------------------------

301  {
302  ++value._data->refCount;
303  if ( --_data->refCount == 0 )
304  delete _data;
305  _data = value._data;
306  return *this;
307  }
