716 class CGE_API cgShaderIdentifier
717 {
718 public:
719  //-------------------------------------------------------------------------
720  // Public Structures, Typedefs & Enumerations
721  //-------------------------------------------------------------------------
722  struct SourceFileInfo
723  {
725  cgString name;
727  cgUInt32 hash[5];
728  };
729  CGE_ARRAY_DECLARE( SourceFileInfo, SourceArray )
730 
731  //-------------------------------------------------------------------------
732  // Constructors & Destructors
733  //-------------------------------------------------------------------------
734  cgShaderIdentifier() {}
735  cgShaderIdentifier( const cgShaderIdentifier * init )
736  {
737  if ( init )
738  {
739  sourceFiles = init->sourceFiles;
740  shaderIdentifier = init->shaderIdentifier;
741  parameterData = init->parameterData;
742  memcpy( inputSignatureHash, init->inputSignatureHash, 5 * sizeof(cgUInt32) );
743 
744  } // End if clone
745  else
746  {
747  memset( inputSignatureHash, 0, 5 * sizeof(cgUInt32) );
748 
749  } // End if !clone
750 
751  } // End Constructor
752 
753  //-------------------------------------------------------------------------
754  // Public Methods
755  //-------------------------------------------------------------------------
756  cgString generateHashName() const;
757 
758  //-------------------------------------------------------------------------
759  // Public Inline Methods
760  //-------------------------------------------------------------------------
761  inline cgInt compare(const cgShaderIdentifier& key2) const
762  {
763  // Check simple count parameters first (fastest first).
764  size_t sourceSize = sourceFiles.size();
765  cgInt difference = ((cgInt)key2.sourceFiles.size() - (cgInt)sourceSize);
766  if ( difference ) return difference;
767  size_t paramSize = parameterData.size();
768  difference = ((cgInt)key2.parameterData.size() - (cgInt)paramSize);
769  if ( difference ) return difference;
770 
771  // Compare shader identifier string (if any).
772  difference = shaderIdentifier.compare( key2.shaderIdentifier );
773  if ( difference ) return difference;
774 
775  // Check parameter contents
776  if ( paramSize )
777  {
778  difference = memcmp( &key2.parameterData[0], &parameterData[0], paramSize * sizeof(cgUInt32) );
779  if ( difference ) return difference;
780 
781  } // End if has parameters
782 
783  // Finally, check source hashes.
784  for ( size_t i = 0; i < sourceSize; ++i )
785  {
786  difference = memcmp( key2.sourceFiles[i].hash, sourceFiles[i].hash, 20 );
787  if ( difference ) return difference;
788 
789  } // Next Source
790 
791  // Identifiers match (for the purpose of comparison)
792  return 0;
793  }
794 
795  //-------------------------------------------------------------------------
796  // Public Inline Operators
797  //-------------------------------------------------------------------------
798  inline bool operator < ( const cgShaderIdentifier & key2 ) const
799  {
800  return (compare( key2 ) < 0);
801  }
802 
803  //-------------------------------------------------------------------------
804  // Public Variables
805  //-------------------------------------------------------------------------
807  SourceArray sourceFiles;
809  cgString shaderIdentifier;
811  cgUInt32Array parameterData;
813  cgUInt32 inputSignatureHash[5];
814 
815 }; // End Class cgShaderIdentifier
