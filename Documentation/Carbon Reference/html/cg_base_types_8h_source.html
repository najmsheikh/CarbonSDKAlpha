147 struct CGE_API cgHalf
148 {
149 public:
150  cgHalf() {};
151  cgHalf( cgFloat f )
152  {
153  const cgUInt32 data = (cgUInt32&)f;
154  const cgUInt32 signBit = (data >> 31);
155  cgUInt32 exponent = ((data >> 23) & 0xFF);
156  cgUInt32 mantissa = (data & 0x7FFFFF);
157 
158  // Handle cases
159  if ( exponent == 255 ) // NaN or inf
160  exponent = 31;
161  else if ( exponent < 102 ) // (127-15)-10
162  exponent = mantissa = 0;
163  else if ( exponent >= 143 ) // 127+(31-15)
164  {
165  exponent = 31;
166  mantissa = 0;
167  }
168  else if( exponent <= 112 ) // 127-15
169  {
170  mantissa |= (1<<23);
171  mantissa = mantissa >> (1 + (112 - exponent));
172  exponent = 0;
173  }
174  else
175  exponent -= 112;
176 
177  // Store
178  _data = (cgUInt16)((signBit << 15) | (exponent << 10) | (mantissa >> 13));
179  }
180 
181  cgHalf( const cgHalf & h ) :
182  _data( h._data ) {}
183 
184  // Inline operators
185  inline bool operator == ( const cgHalf & h ) const
186  {
187  return _data == h._data;
188  }
189  inline bool operator != ( const cgHalf & h ) const
190  {
191  return _data != h._data;
192  }
193  inline operator cgFloat() const
194  {
195  const cgUInt32 signBit = (_data >> 15);
196  cgUInt32 exponent = ((_data >> 10) & 0x1F);
197  cgUInt32 mantissa = (_data & 0x03FF);
198 
199  // Handle cases
200  if(exponent == 31)
201  exponent = 255;
202  else if(exponent == 0)
203  exponent = 0;
204  else
205  exponent += 112;
206 
207  // Convert
208  const cgUInt32 data = (signBit << 31) | (exponent << 23) | (mantissa << 13);
209  return (cgFloat&)data;
210  }
211 
212 private:
213  // Private Members
214  cgUInt16 _data;
215 
216 }; // End Struct cgHalf
